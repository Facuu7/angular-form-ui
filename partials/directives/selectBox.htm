selectBox Go to documentation

This directive gives developers a fancy way to present select boxes.

{{ selectedValue.hexCode }}

Javascript

controller('ExampleController', function ($scope) {
    $scope.options = [{
        name: 'Blue',
        hexCode: '#0000ff'
    },{
        name: 'Red',
        hexCode: '#C41A16'
    },{
        name: 'Purple',
        hexCode: '#AA0D91'
    }];
    $scope.selectedValue;
});

HTML

<select-box ng-model="selectedValue" optExp="t.name for t in options"></select-box>
