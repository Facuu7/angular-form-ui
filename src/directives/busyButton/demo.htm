busyButton

Handling of the "busy" status of a form button

//javascript
controller('ExampleController', function ($scope, $timeout, $q) {
    $scope.save = function () {
        var deferred = $q.defer();
        $timeout(function () {
            //do something here
            deferred.resolve();
        }, 2000);
        return deferred.promise;
    }
});

//html
<button busy-button="Saving..." ng-click="save()"></button>
